```python
# auto_engineering/gates/deep_audit.py
"""v5.5 — DeepAuditGate: 全量代码深度审计 Gate.

与 AuditGate 互补:
  - AuditGate: grep-style 快速扫描 (确定性, <1s)
  - DeepAuditGate: LLM 3-agent 并行审计 (30-60s, critic APPROVE 后触发)

触发时机: Orchestrator 在 critic APPROVE + gates passed 后调
  DeepAuditGate.run() (B7.1 步2j) → GateVerdict, 若 failed (P0>0 或 P1>阈值)
  → T9 plan-refine 回路回到 architect.

输出: GateVerdict 含 P0/P1/P2 分类 findings.
"""

from __future__ import annotations

import logging
from dataclasses import dataclass, field
from pathlib import Path

from auto_engineering.gates.base import Gate, GateVerdict

_logger = logging.getLogger("ae.gates.deep_audit")

# 默认 P1 阈值 — 从 6 开始, 运行后自动学习调整
DEFAULT_P1_THRESHOLD = 6
DEFAULT_MAX_P0 = 0


@dataclass
class DeepAuditFinding:
    """深度审计发现 (比 AuditFinding 多 suggested_fix + agent_source)."""

    severity: str  # "P0" | "P1" | "P2"
    dimension: str  # "架构合理性" | "代码质量" | "工程化规范" | "团队协作友好度" | "代码逻辑虚化度"
    file: str
    line: int
    description: str
    evidence: str = ""
    suggested_fix: str = ""
    agent_source: str = ""  # "arch_quality" | "code_eng" | "collab"


class DeepAuditGate(Gate):
    """全量代码深度审计 Gate (v5.5).

    在 critic APPROVE 后触发, spawn 3 个并行 agent 做 5 维度审计。
    判定: P0>0 或 P1>p1_threshold → failed (触发 T9 plan-refine 回路).

    Args:
        p1_threshold: P1 数量上限 (默认 6, 可从 audit-history.jsonl 学习)
        max_p0: P0 数量上限 (默认 0)
    """

    name = "deep_audit"
    applies_to_stages = ("critic",)  # 仅在 critic APPROVE 后触发

    def __init__(self, p1_threshold: int = DEFAULT_P1_THRESHOLD, max_p0: int = DEFAULT_MAX_P0):
        self.p1_threshold = p1_threshold
        self.max_p0 = max_p0

    def run(self, project_root: Path, contracts: dict | None = None) -> GateVerdict:
        """执行深度审计 (Phase 1: 返回空结果占位, Phase 2 接入 3-agent).

        Args:
            project_root: 项目根目录
            contracts: 可选契约 dict (增量扫描用, DeepAuditGate 全量扫描暂不使用)

        Returns:
            GateVerdict: passed=True 或 passed=False + 结构化 findings
        """
        project_root = Path(project_root)
        if not project_root.exists():
            return GateVerdict.failed(
                f"project_root 不存在: {project_root}",
                gate_name=self.name,
            )

        # Phase 1: 占位 — 不 spawn agent, 返回 passed
        # Phase 2: 接入 3-agent 并行编排器 (loop/deep_audit.py)
        _logger.info("DeepAuditGate.run() — Phase 1 skeleton, returning passed")
        return GateVerdict.passed(
            "DeepAudit Phase 1 skeleton — 3-agent 审计待 Phase 2 接入",
            gate_name=self.name,
        )

    def _build_verdict(
        self, findings: list[DeepAuditFinding], files_scanned: int
    ) -> GateVerdict:
        """从 findings 构建 GateVerdict (Phase 2 接入 3-agent 后使用)."""
        p0 = [f for f in findings if f.severity == "P0"]
        p1 = [f for f in findings if f.severity == "P1"]
        p2 = [f for f in findings if f.severity == "P2"]

        failed = len(p0) > self.max_p0 or len(p1) > self.p1_threshold

        msg_lines = [
            f"DeepAudit 完成: {files_scanned} 文件扫描, "
            f"P0={len(p0)} (max={self.max_p0}), "
            f"P1={len(p1)} (threshold={self.p1_threshold}), "
            f"P2={len(p2)}",
        ]

        if p0:
            msg_lines.append(f"\n  P0 ({len(p0)}):")
            for f in p0[:5]:
                msg_lines.append(f"    [{f.dimension}] {f.file}:{f.line} — {f.description}")
        if p1:
            msg_lines.append(f"\n  P1 ({len(p1)}):")
            for f in p1[:5]:
                msg_lines.append(f"    [{f.dimension}] {f.file}:{f.line} — {f.description}")
        if p2:
            msg_lines.append(f"\n  P2 ({len(p2)}):")
            for f in p2[:3]:
                msg_lines.append(f"    [{f.dimension}] {f.file}:{f.line} — {f.description}")

        if not findings:
            msg_lines.append("\n  无深度审计发现")

        message = "\n".join(msg_lines)

        if failed:
            return GateVerdict.failed(message, gate_name=self.name)
        return GateVerdict.passed(message, gate_name=self.name)
```

"""v5.5 — 3-Agent 并行深度审计编排器.

在 critic APPROVE 后由 Orchestrator._run_deep_audit() 调用.
spawn 3 个并行 agent (arch_quality / code_eng / collab),
收集 findings → 合并去重 → 返回 GateVerdict.

Superpowers 整合: 3 个 agent 的审查 prompt 复用
code-reviewer.md 的 Strengths/Issues/Assessment 输出格式.
"""

from __future__ import annotations

import asyncio
import json
import logging
import os
from dataclasses import dataclass, field
from pathlib import Path
from typing import Any

from auto_engineering.gates.deep_audit import DeepAuditFinding

_logger = logging.getLogger("ae.loop.deep_audit")

# 3 个 agent 的维度分配 (设计文档 B6.5b)
AGENT_DIMENSIONS = {
    "arch_quality": {
        "name": "架构合理性 + 代码逻辑虚化度",
        "dimensions": ["架构合理性", "代码逻辑虚化度"],
        "prompt_file": "~/.claude/superpowers/skills/requesting-code-review/code-reviewer.md",
    },
    "code_eng": {
        "name": "代码质量 + 工程化规范",
        "dimensions": ["代码质量", "工程化规范"],
        "prompt_file": "~/.claude/superpowers/skills/requesting-code-review/code-reviewer.md",
    },
    "collab": {
        "name": "团队协作友好度",
        "dimensions": ["团队协作友好度"],
        "prompt_file": "~/.claude/superpowers/skills/requesting-code-review/code-reviewer.md",
    },
}


@dataclass
class AuditAgentResult:
    """单个审计 agent 的返回结果."""
    agent_id: str
    findings: list[DeepAuditFinding] = field(default_factory=list)
    files_scanned: int = 0
    error: str | None = None


def build_audit_agent_prompt(
    agent_id: str,
    project_root: Path,
    files_changed: list[str] | None = None,
) -> str:
    """为指定审计 agent 构建 prompt (复用 Superpowers code-reviewer.md 模板).

    Args:
        agent_id: "arch_quality" | "code_eng" | "collab"
        project_root: 项目根目录
        files_changed: 本轮变更文件 (可选, 供 agent 聚焦)

    Returns:
        完整 agent prompt 字符串
    """
    dims = AGENT_DIMENSIONS[agent_id]

    # 映射到 code-reviewer.md 输出格式
    prompt = f"""You are a Senior Code Reviewer specializing in: {dims['name']}.

## Audit Scope
Project: {project_root}
Dimensions: {', '.join(dims['dimensions'])}
"""

    if files_changed:
        prompt += f"\nFocus files (recent changes): {', '.join(files_changed[:20])}\n"

    prompt += f"""
## What to Check

{_get_dimension_checklist(agent_id)}

## Calibration

Categorize issues by actual severity. Not everything is Critical.
Acknowledge what was done well before listing issues.

## Output Format

Return a JSON object with this structure:
{{
  "strengths": ["specific strength 1", "specific strength 2"],
  "issues": {{
    "critical": [
      {{"file": "path:line", "description": "...", "evidence": "...", "suggested_fix": "..."}}
    ],
    "important": [
      {{"file": "path:line", "description": "...", "evidence": "...", "suggested_fix": "..."}}
    ],
    "minor": [
      {{"file": "path:line", "description": "...", "evidence": "...", "suggested_fix": "..."}}
    ]
  }},
  "assessment": {{
    "verdict": "PASS|FAIL",
    "reasoning": "1-2 sentence technical assessment"
  }}
}}

Critical → P0 severity
Important → P1 severity
Minor → P2 severity
"""
    return prompt


def _get_dimension_checklist(agent_id: str) -> str:
    """返回每个 agent 的审计维度清单 (复用 /audit skill Phase 2 核查清单)."""
    checklists = {
        "arch_quality": """**架构合理性:**
- 模块边界清晰、职责单一
- 无循环依赖
- God Class/Function (>300行或>20方法)
- 设计 vs 实现差异
- 过度设计 (单消费者接口/抽象类)

**代码逻辑虚化度:**
- 声明的钩子/回调从未赋值
- Public 函数零调用方
- 导出但模块内不存在的符号
- 类实现接口但从未通过接口类型使用""",

        "code_eng": """**代码质量:**
- 异常处理完整 (无静默吞异常)
- 边界条件覆盖 (None/null/空)
- 竞态条件 (async 无锁共享状态)
- 资源泄漏 (文件句柄/连接/定时器)

**工程化规范:**
- 命名一致性

def merge_findings(results: list[AuditAgentResult]) -> list[DeepAuditFinding]:
    """合并 3 个 agent 的 findings, 去重 + severity 归一化.

    3-agent 输出格式使用 Critical/Important/Minor 标签 (B6.5b 模板).
    本函数将其归一化为 P0/P1/P2, 然后按 file:line 去重.

    Args:
        results: 3 个 agent 的 AuditAgentResult

    Returns:
        合并去重后的 DeepAuditFinding 列表
    """
    severity_order = {"P0": 3, "P1": 2, "P2": 1}
    # LLM 输出 severity 归一化: Critical→P0, Important→P1, Minor→P2
    _severity_norm = {"Critical": "P0", "Important": "P1", "Minor": "P2",
                      "critical": "P0", "important": "P1", "minor": "P2"}
    seen: dict[str, DeepAuditFinding] = {}

    for result in results:
        if result.error:
            _logger.warning("Agent %s error: %s", result.agent_id, result.error)
            continue
        for f in result.findings:
            # 归一化 severity (LLM 可能输出 Critical 或 P0)
            if f.severity in _severity_norm:
                f.severity = _severity_norm[f.severity]
            key = f"{f.file}:{f.line}:{f.dimension}"
            if key in seen:
                # 保留更高 severity
                if severity_order.get(f.severity, 0) > severity_order.get(seen[key].severity, 0):
                    seen[key] = f
            else:
                seen[key] = f

    return list(seen.values())


def findings_to_verdict_dict(findings: list[DeepAuditFinding]) -> dict[str, Any]:
    """将 findings 转换为可序列化的 dict (供 JSONL 日志 + GateVerdict 构造)."""
    p0 = [f for f in findings if f.severity == "P0"]
    p1 = [f for f in findings if f.severity == "P1"]
    p2 = [f for f in findings if f.severity == "P2"]
    return {
        "p0_count": len(p0),
        "p1_count": len(p1),
        "p2_count": len(p2),
        "findings": [
            {
                "severity": f.severity,
                "dimension": f.dimension,
                "file": f.file,
                "line": f.line,
                "description": f.description,
                "evidence": f.evidence,
                "suggested_fix": f.suggested_fix,
                "agent_source": f.agent_source,
            }
            for f in findings
        ],
    }

"""v5.5 — 审计历史 JSONL 日志.

供 P1 阈值学习器和 max_iter 学习器读取历史数据.

格式 (B6.5c):
  {"timestamp":"ISO8601","round_id":N,"p0_count":N,"p1_count":N,
   "p2_count":N,"files_scanned":N,"p1_threshold_used":N,"verdict":"passed|failed"}

与 run-history.jsonl (max_iter 学习):
  {"timestamp":"ISO8601","requirement":"...","rounds_completed":N,
   "convergence_verdict":"GOAL_ACHIEVED|MAX_ITERATIONS|...",
   "gate_results":{"safety":"passed",...},"duration_seconds":N}

from __future__ import annotations

import json
import logging
from datetime import datetime, timezone
from pathlib import Path
from typing import Any

_logger = logging.getLogger("ae.loop.audit_history")

AE_STATE_DIR = ".ae-state"
AUDIT_HISTORY_FILE = "audit-history.jsonl"
RUN_HISTORY_FILE = "run-history.jsonl"


def _ensure_ae_state(project_root: Path) -> Path:
    """确保 .ae-state/ 目录存在."""
    state_dir = project_root / AE_STATE_DIR
    state_dir.mkdir(parents=True, exist_ok=True)
    return state_dir


def append_audit_entry(
    project_root: Path,
    round_id: int,
    p0_count: int,
    p1_count: int,
    p2_count: int,
    files_scanned: int,
    p1_threshold_used: int,
    verdict: str,
) -> None:
    """追加一行审计日志到 audit-history.jsonl."""
    state_dir = _ensure_ae_state(project_root)
    entry = {
        "timestamp": datetime.now(timezone.utc).isoformat(),
        "round_id": round_id,
        "p0_count": p0_count,
        "p1_count": p1_count,
        "p2_count": p2_count,
        "files_scanned": files_scanned,
        "p1_threshold_used": p1_threshold_used,
        "verdict": verdict,
    }
    log_path = state_dir / AUDIT_HISTORY_FILE
    with open(log_path, "a", encoding="utf-8") as f:
        f.write(json.dumps(entry, ensure_ascii=False) + "\n")
    _logger.debug("Audit entry written: %s", entry)

    project_root: Path,
    requirement: str,
    rounds_completed: int,
    convergence_verdict: str,

    _logger.debug("Run entry written: rounds=%d verdict=%s", rounds_completed, convergence_verdict)


def read_audit_history(project_root: Path) -> list[dict[str, Any]]:
    """读取全部审计历史记录."""
    log_path = project_root / AE_STATE_DIR / AUDIT_HISTORY_FILE
    if not log_path.exists():
        return []
    entries: list[dict[str, Any]] = []
    with open(log_path, "r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if line:
                try:
                    entries.append(json.loads(line))
                except json.JSONDecodeError:
                    _logger.warning("Skipping malformed audit history line")
    return entries

- [ ] **Step 1: 实现学习算法 (设计文档 B6.5c + B6.5d)**

```python
# auto_engineering/loop/threshold_learner.py
"""v5.5 — P1 阈值 + max_iter 自动学习.

设计来源: design/v5.0-Design-Loop.md §B6.5c + §B6.5d

P1 阈值学习 (B6.5c):
  - 从 audit-history.jsonl 读取历史 P1 数据
  - 计算 median / p75 / p90 / max
  - 推荐 = p75 (避免过多 false positive 又不放过真实问题)
  - 连续 3 次一致推荐 → 自动调整

max_iter 学习 (B6.5d):
  - 从 run-history.jsonl 读取历史收敛轮次
  - 推荐 = min(avg_rounds × 2, 20)
  - 质量-成本检查: 若 90% 运行在 avg_rounds 内收敛 → 推荐合理
"""

from __future__ import annotations

import logging
import statistics
from dataclasses import dataclass
from pathlib import Path

from auto_engineering.loop.audit_history import read_audit_history, read_run_history

_logger = logging.getLogger("ae.loop.threshold_learner")

DEFAULT_P1_THRESHOLD = 6
MIN_SAMPLES_FOR_LEARNING = 5  # 最少 5 次运行才开始学习
CONSISTENT_RECOMMENDATIONS = 3  # 连续 3 次一致推荐 → 自动调整


@dataclass
class ThresholdRecommendation:
    """P1 阈值推荐."""

    recommended: int
    median: float
    p75: float
    p90: float
    max_val: int
    sample_count: int
    confidence: str  # "high" | "medium" | "low" | "insufficient_data"


@dataclass
class MaxIterRecommendation:
    """max_iter 推荐."""

    recommended: int
    avg_rounds: float
    median_rounds: float
    p90_rounds: float
    max_rounds: int
    sample_count: int
    convergence_rate: float  # GOAL_ACHIEVED / total
    quality_cost_ok: bool  # 质量-成本平衡检查


def recommend_p1_threshold(project_root: Path) -> ThresholdRecommendation:
    """从 audit-history.jsonl 计算 P1 阈值推荐.

    算法:
      1. 收集历史 P1 数据
      2. 计算 median / p75 / p90 / max
      3. 推荐 = p75 (取整)
      4. 样本不足 → 返回默认 6 + confidence=insufficient_data
    """
    history = read_audit_history(project_root)
    p1_values = [e["p1_count"] for e in history if "p1_count" in e]

    if len(p1_values) < MIN_SAMPLES_FOR_LEARNING:
        return ThresholdRecommendation(
            recommended=DEFAULT_P1_THRESHOLD,
            median=0.0,
            p75=0.0,
            p90=0.0,
            max_val=0,
            sample_count=len(p1_values),
            confidence="insufficient_data",
        )

    sorted_values = sorted(p1_values)
    med = float(statistics.median(sorted_values))

    # p75: 第 75 百分位
    p75_idx = int(len(sorted_values) * 0.75)
    p75 = float(sorted_values[min(p75_idx, len(sorted_values) - 1)])

    # p90: 第 90 百分位
    p90_idx = int(len(sorted_values) * 0.90)
    p90 = float(sorted_values[min(p90_idx, len(sorted_values) - 1)])

    max_val = max(sorted_values)
    recommended = max(int(p75), 1)  # 至少 1

    # 置信度判定
    if len(p1_values) >= 10:
        confidence = "high"
    elif len(p1_values) >= 7:
        confidence = "medium"
    else:
        confidence = "low"

    return ThresholdRecommendation(
        recommended=recommended,
        median=med,
        p75=p75,
        p90=p90,
        max_val=max_val,
        sample_count=len(p1_values),
        confidence=confidence,
    )


def recommend_max_iter(project_root: Path) -> MaxIterRecommendation:
    """从 run-history.jsonl 计算 max_iter 推荐.

    算法:
      1. 收集历史 rounds_completed
      2. 计算 avg / median / p90 / max
      3. 推荐 = min(avg_rounds × 2, 20)
      4. 质量-成本检查: ≥80% 运行在 avg×2 内收敛 → OK
    """
    history = read_run_history(project_root)
    rounds = [
        e["rounds_completed"]
        for e in history
        if "rounds_completed" in e and isinstance(e["rounds_completed"], (int, float))
    ]

    if len(rounds) < MIN_SAMPLES_FOR_LEARNING:
        return MaxIterRecommendation(
            recommended=10,  # 默认 10
            avg_rounds=0.0,
            median_rounds=0.0,
            p90_rounds=0.0,
            max_rounds=0,
            sample_count=len(rounds),
            convergence_rate=0.0,
            quality_cost_ok=True,
        )

    avg = float(statistics.mean(rounds))
    med = float(statistics.median(rounds))

    sorted_rounds = sorted(rounds)
    p90_idx = int(len(sorted_rounds) * 0.90)
    p90 = float(sorted_rounds[min(p90_idx, len(sorted_rounds) - 1)])
    max_r = max(sorted_rounds)

    recommended = min(int(avg * 2), 20)
    recommended = max(recommended, 3)  # 至少 3

    # 质量-成本检查: 多少比例在 avg×2 内收敛
    within_budget = sum(1 for r in rounds if r <= avg * 2)
    within_budget_rate = within_budget / len(rounds)
    quality_cost_ok = within_budget_rate >= 0.80

    # 收敛率: GOAL_ACHIEVED / total
    converged = sum(
        1
        for e in history
        if e.get("convergence_verdict") == "GOAL_ACHIEVED"
    )
    convergence_rate = converged / len(history) if history else 0.0

    return MaxIterRecommendation(
        recommended=recommended,
        avg_rounds=avg,
        median_rounds=med,
        p90_rounds=p90,
        max_rounds=max_r,
        sample_count=len(rounds),
        convergence_rate=convergence_rate,
        quality_cost_ok=quality_cost_ok,
    )


def should_auto_adjust(
    recommendations: list[int], current_threshold: int
) -> tuple[bool, int | None]:
    """判断是否应自动调整阈值.

    条件: 连续 CONSISTENT_RECOMMENDATIONS 次推荐相同值且 ≠ current_threshold.

    Returns:
        (should_adjust, new_value_or_None)
    """
    if len(recommendations) < CONSISTENT_RECOMMENDATIONS:
        return False, None

    recent = recommendations[-CONSISTENT_RECOMMENDATIONS:]
    if len(set(recent)) == 1 and recent[0] != current_threshold:
        return True, recent[0]

    return False, None
```

def _run_deep_audit(self) -> GateVerdict:
    """v5.5: 运行 DeepAuditGate 全量代码审计.

    critic APPROVE + 6 Gate passed 后调用.
    判定: P0>0 或 P1>threshold → failed (触发 T9 plan-refine 回路).

    Returns:
        GateVerdict: passed 或 failed
    """
    from auto_engineering.gates.deep_audit import DeepAuditGate
    from auto_engineering.loop.audit_history import append_audit_entry
    from auto_engineering.loop.threshold_learner import recommend_p1_threshold

    project_root = Path(self.config.project_root or Path.cwd())

    # 从历史学习 P1 阈值
    rec = recommend_p1_threshold(project_root)
    p1_threshold = rec.recommended if rec.confidence in ("high", "medium") else 6

    gate = DeepAuditGate(p1_threshold=p1_threshold)
    verdict = gate.run(project_root)

    # 记录审计日志
    p0_count = verdict.message.count("P0=")  # TODO: Phase 2 接入 3-agent 后改用结构化数据
    append_audit_entry(
        project_root=project_root,
        round_id=self.state.current_round if hasattr(self.state, 'current_round') else 0,
        p0_count=p0_count,
        p1_count=0,
        p2_count=0,
        files_scanned=0,
        p1_threshold_used=p1_threshold,
        verdict="passed" if verdict.passed else "failed",
    )

    return verdict
